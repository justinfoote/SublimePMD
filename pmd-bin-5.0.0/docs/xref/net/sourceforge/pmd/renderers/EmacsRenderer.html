11  import net.sourceforge.pmd.RuleViolation;

13  /**
14   * Renderer to GNU Emacs parsable format.
15   */
16  public class EmacsRenderer extends AbstractIncrementingRenderer {
17  
18      public static final String NAME = "emacs";
19  
20      protected static final String EOL = System.getProperty("line.separator", "\n");
21  
22      public EmacsRenderer(Properties properties) {
23      	super(NAME, "GNU Emacs integration.", properties);
24      }
25  
26      public String defaultFileExtension() { return "emacs"; }
27      
28      /**
29       * {@inheritDoc}
30       */
31      @Override
32      public void renderFileViolations(Iterator<RuleViolation> violations) throws IOException {
33  		Writer writer = getWriter();
34  		StringBuilder buf = new StringBuilder();
35  		while (violations.hasNext()) {
36  		    RuleViolation rv = violations.next();
37  		    buf.setLength(0);
38  		    buf.append(rv.getFilename());
39  		    buf.append(':').append(Integer.toString(rv.getBeginLine()));
40  		    buf.append(": ").append(rv.getDescription()).append(EOL);
41  		    writer.write(buf.toString());
42  		}
43      }
44  }
