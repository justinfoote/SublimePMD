80       * Enables colors on *nix systems - not windows. Color support depends
81       * on the pmd.color property, which should be set with the -D option
82       * during execution - a set value other than 'false' or '0' enables color.
83       * <p/>
84       * btw, is it possible to do this on windows (ie; console colors)?

86      private void initializeColorsIfSupported() {
87  	if (System.getProperty("pmd.color") != null
88  		&& !(System.getProperty("pmd.color").equals("0") || System.getProperty("pmd.color").equals("false"))) {

102     @Override
103     public void end() throws IOException {
104 	StringBuffer buf = new StringBuffer(500);
105 	buf.append(PMD.EOL);
106 	initializeColorsIfSupported();
107 	String lastFile = null;
108 	int numberOfErrors = 0;
109 	int numberOfWarnings = 0;
110 
111 	for (Iterator<RuleViolation> i = report.iterator(); i.hasNext();) {
112 	    buf.setLength(0);
113 	    numberOfWarnings++;
114 	    RuleViolation rv = i.next();
115 	    if (!rv.getFilename().equals(lastFile)) {
116 		lastFile = rv.getFilename();
117 		buf.append(this.yellowBold + "*" + this.colorReset + " file: " + this.whiteBold
118 			+ this.getRelativePath(lastFile) + this.colorReset + PMD.EOL);
119 	    }
120 	    buf.append(this.green + "    src:  " + this.cyan
121 		    + lastFile.substring(lastFile.lastIndexOf(File.separator) + 1) + this.colorReset + ":" + this.cyan
122 		    + rv.getBeginLine() + (rv.getEndLine() == -1 ? "" : ":" + rv.getEndLine()) + this.colorReset
123 		    + PMD.EOL);
124 	    buf.append(this.green + "    rule: " + this.colorReset + rv.getRule().getName() + PMD.EOL);
125 	    buf.append(this.green + "    msg:  " + this.colorReset + rv.getDescription() + PMD.EOL);
126 	    buf.append(this.green + "    code: " + this.colorReset + this.getLine(lastFile, rv.getBeginLine())
127 		    + PMD.EOL + PMD.EOL);
128 	    writer.write(buf.toString());
129 	}
130 	writer.write(PMD.EOL + PMD.EOL);
131 	writer.write("Summary:" + PMD.EOL + PMD.EOL);
132 	Map<String, Integer> summary = report.getCountSummary();
133 	for (Map.Entry<String, Integer> entry : summary.entrySet()) {
134 	    buf.setLength(0);
135 	    String key = entry.getKey();
136 	    buf.append(key).append(" : ").append(entry.getValue()).append(PMD.EOL);
137 	    writer.write(buf.toString());
138 	}
139 
140 	for (Iterator<Report.ProcessingError> i = report.errors(); i.hasNext();) {
141 	    buf.setLength(0);
142 	    numberOfErrors++;
143 	    Report.ProcessingError error = i.next();
144 	    if (error.getFile().equals(lastFile)) {
145 		lastFile = error.getFile();
146 		buf.append(this.redBold + "*" + this.colorReset + " file: " + this.whiteBold
147 			+ this.getRelativePath(lastFile) + this.colorReset + PMD.EOL);
148 	    }
149 	    buf.append(this.green + "    err:  " + this.cyan + error.getMsg() + this.colorReset + PMD.EOL + PMD.EOL);
150 	    writer.write(buf.toString());
151 	}
152 
153 	// adding error message count, if any
154 	if (numberOfErrors > 0) {
155 	    writer.write(this.redBold + "*" + this.colorReset + " errors:   " + this.whiteBold + numberOfWarnings
156 		    + this.colorReset + PMD.EOL);
157 	}
158 	writer.write(this.yellowBold + "*" + this.colorReset + " warnings: " + this.whiteBold + numberOfWarnings
159 		+ this.colorReset + PMD.EOL);
160     }

162     /**
163      * Retrieves the requested line from the specified file.
164      *
165      * @param sourceFile the java or cpp source file
166      * @param line       line number to extract
167      * @return a trimmed line of source code
168      */
169     private String getLine(String sourceFile, int line) {
170 		String code = null;
171 		BufferedReader br = null;
172 		try {
173 		    br = new BufferedReader(getReader(sourceFile));
174 		    for (int i = 0; line > i; i++) {
175 		    	String txt = br.readLine();
176 				code = txt == null ? "" : txt.trim();
177 			    }
178 		} catch (IOException ioErr) {
179 		    ioErr.printStackTrace();
180 		} finally {
181 			IOUtil.closeQuietly(br);
182 		}
183 		return code;
184     }

190     /**
191      * Attempts to determine the relative path to the file. If relative path cannot be found,
192      * the original path is returnedi, ie - the current path for the supplied file.
193      *
194      * @param fileName well, the file with its original path.
195      * @return the relative path to the file
196      */
197     private String getRelativePath(String fileName) {
198 	String relativePath;
199 
200 	// check if working directory need to be assigned
201 	if (pwd == null) {
202 	    try {
203 		this.pwd = new File(".").getCanonicalPath();
204 	    } catch (IOException ioErr) {
205 		// to avoid further error
206 		this.pwd = "";
207 	    }
208 	}
209 
210 	// make sure that strings match before doing any substring-ing
211 	if (fileName.indexOf(this.pwd) == 0) {
212 	    relativePath = "." + fileName.substring(this.pwd.length());
213 
214 	    // remove current dir occuring twice - occurs if . was supplied as path
215 	    if (relativePath.startsWith("." + File.separator + "." + File.separator)) {
216 		relativePath = relativePath.substring(2);
217 	    }
218 	} else {
219 	    // this happens when pmd's supplied argument deviates from the pwd 'branch' (god knows this terminolgy - i hope i make some sense).
220 	    // for instance, if supplied=/usr/lots/of/src and pwd=/usr/lots/of/shared/source
221 	    // TODO: a fix to get relative path?
222 	    relativePath = fileName;
223 	}
224 
225 	return relativePath;
226     }
