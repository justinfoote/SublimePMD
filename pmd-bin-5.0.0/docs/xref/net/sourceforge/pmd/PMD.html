74  	public static Report setupReport(RuleSets rs, RuleContext ctx, String fileName) {
75  		
76  		Set<Rule> brokenRules = removeBrokenRules(rs);
77  		Report report = Report.createReport(ctx, fileName);
78  		
79  		for (Rule rule : brokenRules) {
80  			report.addConfigError(
81  				new Report.RuleConfigurationError(rule, rule.dysfunctionReason())
82  				);
83  		}
84  
85  		return report;

99      	ruleSets.removeDysfunctionalRules(brokenRules);
100

146 	/** 
147 	 * This method is the main entry point for command line usage.
148 	 * 
149 	 * @param configuration
150 	 */
151 	public static void doPMD(PMDConfiguration configuration) {
152 
153 		// Load the RuleSets
154 		long startLoadRules = System.nanoTime();
155 		RuleSetFactory ruleSetFactory = RulesetsFactoryUtils
156 				.getRulesetFactory(configuration);
157 
158 		RuleSets ruleSets = RulesetsFactoryUtils.getRuleSets(
159 				configuration.getRuleSets(), ruleSetFactory, startLoadRules);
160 		if (ruleSets == null)
161 			return;
162 
163 		Set<Language> languages = getApplicableLanguages(configuration, ruleSets);
164 		List<DataSource> files = getApplicableFiles(configuration, languages);
165 
166 		long reportStart = System.nanoTime();
167 		try {			
168 			Renderer renderer = configuration.createRenderer();
169 			List<Renderer> renderers = new LinkedList<Renderer>();
170 			renderers.add(renderer);
171 
172 			renderer.setWriter(IOUtil.createWriter(configuration.getReportFile()));
173 			renderer.start();
174 
175 			Benchmarker.mark(Benchmark.Reporting, System.nanoTime() - reportStart, 0);
176 
177 			RuleContext ctx = new RuleContext();
178 
179 			processFiles(configuration, ruleSetFactory, files, ctx, renderers);
180 
181 			reportStart = System.nanoTime();
182 			renderer.end();
183 			renderer.flush();
184 		} catch (Exception e) {
185 			String message = e.getMessage();
186 			if (message != null) {
187 				LOG.severe(message);
188 			} else {
189 				LOG.log(Level.SEVERE, "Exception during processing", e);
190 			}
191 			LOG.log(Level.FINE, "Exception during processing", e);
192 			LOG.info(CommandLineParser.usage());
193 		} finally {
194 			Benchmarker.mark(Benchmark.Reporting, System.nanoTime() - reportStart, 0);
195 		}
196 	}

244 	 * Run PMD on a list of files using multiple threads - if more than one is available
245 	 * 
246 	 * @param configuration Configuration
247 	 * @param ruleSetFactory RuleSetFactory
248 	 * @param files List<DataSource>
249 	 * @param ctx RuleContext
250 	 * @param renderers List<Renderer>
251 	 */
252 	public static void processFiles(final PMDConfiguration configuration,
253 			final RuleSetFactory ruleSetFactory, final List<DataSource> files,
254 			final RuleContext ctx, final List<Renderer> renderers) {
255 
256 		sortFiles(configuration, files);
257 
258 		/*
259 		 * Check if multithreaded support is available. ExecutorService can also be
260 		 * disabled if threadCount is not positive, e.g. using the "-threads 0"
261 		 * command line option.
262 		 */
263 		if (configuration.getThreads() > 0) {
264 			new MultiThreadProcessor(configuration).processFiles(ruleSetFactory, files, ctx, renderers);
265 		} else {
266 			new MonoThreadProcessor(configuration).processFiles(ruleSetFactory, files, ctx, renderers);
267 		}
268 	}

270 	/**
271 	 *
272 	 * @param configuration Configuration
273 	 * @param files List<DataSource>
274 	 */
275 	private static void sortFiles(final PMDConfiguration configuration, final List<DataSource> files) {
276 		if (configuration.isStressTest()) {
277 			// randomize processing order
278 			Collections.shuffle(files);
279 		} else {
280 			final boolean useShortNames = configuration.isReportShortNames();
281 			final String inputPaths = configuration.getInputPaths();
282 			Collections.sort(files, new Comparator<DataSource>() {
283 				public int compare(DataSource left, DataSource right) {
284 					String leftString = left.getNiceFileName(useShortNames, inputPaths);
285 					String rightString = right.getNiceFileName(useShortNames, inputPaths);
286 					return leftString.compareTo(rightString);
287 				}
288 			});
289 		}
290 	}

309 	/**
310 	 *
311 	 * @param configuration Configuration
312 	 * @param ruleSets RuleSets
313 	 * @return Set<Language>
314 	 */
315 	private static Set<Language> getApplicableLanguages(PMDConfiguration configuration, RuleSets ruleSets) {
316 		Set<Language> languages = new HashSet<Language>();
317 		LanguageVersionDiscoverer discoverer = configuration.getLanguageVersionDiscoverer();
318 		
319 		for (Rule rule : ruleSets.getAllRules()) {
320 			Language language = rule.getLanguage();
321 			if (languages.contains(language))
322 				continue;
323 			LanguageVersion version = discoverer.getDefaultLanguageVersion(language);
324 			if (RuleSet.applies(rule, version)) {
325 				languages.add(language);
326 				LOG.fine("Using " + language.getShortName() + 
327 						" version: " + version.getShortName());
328 			}
329 		}
330 		return languages;
331 	}

344     /**
345      *
346      * @param args String[]
347      * @return int
348      */
349     public static int run(String[] args) { 
350     	int status = 0;
351 		long start = System.nanoTime();
352 		final CommandLineParser opts = new CommandLineParser(args);
353 		final PMDConfiguration configuration = opts.getConfiguration();
354 
355 		final Level logLevel = configuration.isDebug() ? Level.FINER : Level.INFO;
356 		final Handler logHandler = new ConsoleLogHandler();
357 		final ScopedLogHandlersManager logHandlerManager = new ScopedLogHandlersManager(logLevel, logHandler);
358 		final Level oldLogLevel = LOG.getLevel();
359 		LOG.setLevel(logLevel); //Need to do this, since the static logger has already been initialized at this point
360 		try {
361 		    PMD.doPMD(opts.getConfiguration());
362 		} catch (Exception e) {
363 			System.out.print(CommandLineParser.usage());
364 			System.out.println(e.getMessage());
365 			status = ERROR_STATUS;
366 		} finally {
367 		    logHandlerManager.close();
368 		    LOG.setLevel(oldLogLevel);
369 		    if (configuration.isBenchmark()) {
370 				long end = System.nanoTime();
371 				Benchmarker.mark(Benchmark.TotalPMD, end - start, 0);
372 				
373 				TextReport report = new TextReport();	// TODO get specified report format from config
374 				report.generate(Benchmarker.values(), System.err);
375 		    }
376 		}
377 		return status;
378     }

394         	}

397     		pmdVersion = "unknown";
398     	}
