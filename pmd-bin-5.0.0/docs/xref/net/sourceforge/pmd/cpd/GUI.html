183     private class SaveListener implements ActionListener {
184     	
185     	final Renderer renderer;
186     	
187     	public SaveListener(Renderer theRenderer) {
188     		renderer = theRenderer;
189     	}
190     	
191         public void actionPerformed(ActionEvent evt) {
192             JFileChooser fcSave	= new JFileChooser();
193             int ret = fcSave.showSaveDialog(GUI.this.frame);
194             File f = fcSave.getSelectedFile();
195             if (f == null || ret != JFileChooser.APPROVE_OPTION) {
196         	return;
197             }
198                         
199             if (!f.canWrite()) {
200                 PrintWriter pw = null;
201                 try {
202                     pw = new PrintWriter(new FileOutputStream(f));
203                     pw.write(renderer.render(matches.iterator()));
204                     pw.flush();
205                     JOptionPane.showMessageDialog(frame, "Saved " + matches.size() + " matches");
206                 } catch (IOException e) {
207                     error("Couldn't save file" + f.getAbsolutePath(), e);
208                 } finally {
209                     IOUtil.closeQuietly(pw);
210                 }
211             } else {
212                 error("Could not write to file " + f.getAbsolutePath(), null);
213             }
214         }
215 
216         private void error(String message, Exception e) {
217             if (e != null) {
218                 e.printStackTrace();
219             }
220             JOptionPane.showMessageDialog(GUI.this.frame, message);
221         }

438     private void copyMatchListSelectionsToClipboard() {
439     	
440     	int[] selectionIndices = resultsTable.getSelectedRows();
441     	int colCount = resultsTable.getColumnCount();
442     	
443     	StringBuilder sb = new StringBuilder();
444     	    	
445     	for (int r=0; r<selectionIndices.length; r++) {
446 			if (r > 0) {
447 			    sb.append('\n');
448 			}
449 			sb.append(resultsTable.getValueAt(selectionIndices[r], 0));
450     		for (int c=1; c<colCount; c++) {
451     			sb.append('\t');
452     			sb.append(resultsTable.getValueAt(selectionIndices[r], c));
453     		}
454     	}
455     	
456     	StringSelection ss = new StringSelection(sb.toString());
457         Toolkit.getDefaultToolkit().getSystemClipboard().setContents(ss, null);
458     }

504     private boolean isLegalPath(String path, LanguageConfig config) {
505     	String[] extensions = config.extensions();
506     	for (int i=0; i<extensions.length; i++) {
507     		if (path.endsWith(extensions[i]) && extensions[i].length() > 0) {
508     		    return true;
509     		}
510     	}
511     	return false;
512     }

514     private String setLabelFor(Match match) {
515     	
516     	Set<String> sourceIDs = new HashSet<String>(match.getMarkCount());
517     	for (Iterator<TokenEntry> occurrences = match.iterator(); occurrences.hasNext();) {
518              sourceIDs.add(occurrences.next().getTokenSrcID());
519           }
520     	String label;
521     	
522     	if (sourceIDs.size() == 1) {
523     		String sourceId = sourceIDs.iterator().next();
524     		int separatorPos = sourceId.lastIndexOf(File.separatorChar);
525     		label = "..." + sourceId.substring(separatorPos);
526     		} else {
527     	    	label = "(" + sourceIDs.size() + " separate files)";
528     		}
529     		
530     	match.setLabel(label);
531     	return label;
532     }

540     private void go() {
541     	String dirPath = rootDirectoryField.getText();
542         try {
543             if (!(new File(dirPath)).exists()) {
544                 JOptionPane.showMessageDialog(frame,
545                         "Can't read from that root source directory",
546                         "Error", JOptionPane.ERROR_MESSAGE);
547                 return;
548             }
549       
550             setProgressControls(true);
551 
552             Properties p = new Properties();
553             p.setProperty(JavaTokenizer.IGNORE_LITERALS, String.valueOf(ignoreLiteralsCheckbox.isSelected()));
554             p.setProperty(LanguageFactory.EXTENSION, extensionField.getText());
555             LanguageConfig conf = languageConfigFor((String)languageBox.getSelectedItem());
556             Language language = conf.languageFor(new LanguageFactory(), p);
557             CPDConfiguration config = new CPDConfiguration(
558             		Integer.parseInt(minimumLengthField.getText()), 
559             		language, encodingField.getText()
560             		);
561             CPD cpd = new CPD(config);
562             cpd.setCpdListener(this);
563             tokenizingFilesBar.setMinimum(0);
564             phaseLabel.setText("");
565             if (isLegalPath(dirPath, conf)) {	// should use the language file filter instead?
566             	cpd.add(new File(dirPath));
567             } else {
568                 if (recurseCheckbox.isSelected()) {
569                     cpd.addRecursively(dirPath);
570                 } else {
571                     cpd.addAllInDirectory(dirPath);
572                 }
573             }
574             Timer t = createTimer();
575             t.start();
576             cpd.go();
577             t.stop();
578             
579         	matches = new ArrayList<Match>();
580         	Match match;
581         	for (Iterator<Match> i = cpd.getMatches(); i.hasNext();) {
582         		match = i.next();
583         		setLabelFor(match);
584         		matches.add(match);
585         	}
586 
587             String report = new SimpleRenderer().render(cpd.getMatches());
588             if (report.length() == 0) {
589                 JOptionPane.showMessageDialog(frame,
590                         "Done; couldn't find any duplicates longer than " + minimumLengthField.getText() + " tokens");
591             } else {
592                 resultsTextArea.setText(report);
593                 setListDataFrom(cpd.getMatches());
594                 
595             }
596         } catch (IOException t) {
597             t.printStackTrace();
598             JOptionPane.showMessageDialog(frame, "Halted due to " + t.getClass().getName() + "; " + t.getMessage());
599         } catch (RuntimeException t) {
600             t.printStackTrace();
601             JOptionPane.showMessageDialog(frame, "Halted due to " + t.getClass().getName() + "; " + t.getMessage());
602         }
603         setProgressControls(false);
604     }

607
