29  	private Document createDocument() {
30      	try {
31  			DocumentBuilderFactory factory = DocumentBuilderFactory
32  					.newInstance();
33  			DocumentBuilder parser = factory.newDocumentBuilder();
34  			return parser.newDocument();
35  		} catch (ParserConfigurationException e) {
36  			throw new IllegalStateException(e);
37  		}
38  	}
39  	
40  	private String xmlDocToString(Document doc) {
41          try {
42  	        TransformerFactory tf = TransformerFactory.newInstance();
43  	        Transformer transformer = tf.newTransformer();
44  	        transformer.setOutputProperty(OutputKeys.ENCODING, System.getProperty("file.encoding"));
45  	        StringWriter writer = new StringWriter();
46  	        transformer.transform(new DOMSource(doc), new StreamResult(writer));
47  	        return writer.getBuffer().toString().replaceAll("\n|\r", "");
48          }  catch (TransformerException e) {
49          	throw new IllegalStateException(e);
50  		}
51  	}

53      public String render(Iterator<Match> matches) {
54      	Document doc = createDocument();
55  		Element root = doc.createElement("pmd-cpd");
56  		doc.appendChild(root);
57  
58          Match match;
59          while (matches.hasNext()) {
60              match = matches.next();
61              root.appendChild( addCodeSnippet(doc, addFilesToDuplicationElement(doc, createDuplicationElement(doc, match), match), match ) );
62          }
63          return xmlDocToString(doc);
64      }
65      
66      private Element addFilesToDuplicationElement(Document doc, Element duplication, Match match) {
67      	TokenEntry mark;
68          for (Iterator<TokenEntry> iterator = match.iterator(); iterator.hasNext();) {
69              mark = iterator.next();
70              Element file = doc.createElement("file");
71              file.setAttribute("line", String.valueOf(mark.getBeginLine()));
72              file.setAttribute("path", mark.getTokenSrcID());
73              duplication.appendChild(file);
74          }
75          return duplication;
76      }
77  
78      private Element addCodeSnippet(Document doc, Element duplication, Match match) {
79          String codeSnipet = match.getSourceCodeSlice();
80          if (codeSnipet != null) {
81          	Element codefragment = doc.createElement("codefragment");
82              codefragment.appendChild(doc.createCDATASection(codeSnipet));
83              duplication.appendChild(codefragment);
84          }
85          return duplication;
86      }
87      
88      private Element createDuplicationElement(Document doc, Match match) {
89          Element duplication = doc.createElement("duplication");
90          duplication.setAttribute("lines", String.valueOf(match.getLineCount()));
91          duplication.setAttribute("tokens", String.valueOf(match.getTokenCount()));
92          return duplication;
93      }
