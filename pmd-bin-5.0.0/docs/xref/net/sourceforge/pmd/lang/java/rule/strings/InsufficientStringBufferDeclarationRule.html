31  /**
32   * This rule finds StringBuffers which may have been pre-sized incorrectly
33   *
34   * See http://sourceforge.net/forum/forum.php?thread_id=1438119&forum_id=188194
35   * @author Allan Caplan
36   */
37  public class InsufficientStringBufferDeclarationRule extends AbstractJavaRule {
38  
39      private final static Set<Class<? extends Node>> BLOCK_PARENTS;
40  
41      static {
42          BLOCK_PARENTS = new HashSet<Class<? extends Node>>(2);
43          BLOCK_PARENTS.add(ASTIfStatement.class);
44          BLOCK_PARENTS.add(ASTSwitchStatement.class);
45      }
46  
47      public static final int DEFAULT_BUFFER_SIZE = 16;	// as specified in StringBuffer & StringBuilder
48  
49      @Override
50      public Object visit(ASTVariableDeclaratorId node, Object data) {
51          if (!TypeHelper.isEither(node.getNameDeclaration(), StringBuffer.class, StringBuilder.class)) {
52              return data;
53          }
54          Node rootNode = node;
55          int anticipatedLength = 0;
56          int constructorLength = DEFAULT_BUFFER_SIZE;
57  
58          constructorLength = getConstructorLength(node, constructorLength);
59          anticipatedLength = getInitialLength(node);
60          List<NameOccurrence> usage = node.getUsages();
61          Map<Node, Map<Node, Integer>> blocks = new HashMap<Node, Map<Node, Integer>>();
62          for (int ix = 0; ix < usage.size(); ix++) {
63              NameOccurrence no = usage.get(ix);
64              Node n = no.getLocation();
65              if (!InefficientStringBufferingRule.isInStringBufferOperation(n, 3, "append")) {
66  
67                  if (!no.isOnLeftHandSide() && !InefficientStringBufferingRule.isInStringBufferOperation(n, 3, "setLength")) {
68                      continue;
69                  }
70                  if (constructorLength != -1 && anticipatedLength > constructorLength) {
71                      anticipatedLength += processBlocks(blocks);
72                      String[] param = { String.valueOf(constructorLength), String.valueOf(anticipatedLength) };
73                      addViolation(data, rootNode, param);
74                  }
75                  constructorLength = getConstructorLength(n, constructorLength);
76                  rootNode = n;
77                  anticipatedLength = getInitialLength(node);
78              }
79              ASTPrimaryExpression s = n.getFirstParentOfType(ASTPrimaryExpression.class);
80              int numChildren = s.jjtGetNumChildren();
81              for (int jx = 0; jx < numChildren; jx++) {
82              	Node sn = s.jjtGetChild(jx);
83                  if (!(sn instanceof ASTPrimarySuffix) || sn.getImage() != null) {
84                      continue;
85                  }
86                  int thisSize = 0;
87                  Node block = getFirstParentBlock(sn);
88                  if (isAdditive(sn)) {
89                      thisSize = processAdditive(sn);
90                  } else {
91                      thisSize = processNode(sn);
92                  }
93                  if (block != null) {
94                      storeBlockStatistics(blocks, thisSize, block);
95                  } else {
96                      anticipatedLength += thisSize;
97                  }
98              }
99          }
100         anticipatedLength += processBlocks(blocks);
101         if (constructorLength != -1 && anticipatedLength > constructorLength) {
102             String[] param = { String.valueOf(constructorLength), String.valueOf(anticipatedLength) };
103             addViolation(data, rootNode, param);
104         }
105         return data;
106     }

108     /**
109      * This rule is concerned with IF and Switch blocks. Process the block into
110      * a local Map, from which we can later determine which is the longest block
111      * inside
112      *
113      * @param blocks
114      *            The map of blocks in the method being investigated
115      * @param thisSize
116      *            The size of the current block
117      * @param block
118      *            The block in question
119      */
120     private void storeBlockStatistics(Map<Node, Map<Node, Integer>> blocks, int thisSize, Node block) {
121         Node statement = block.jjtGetParent();
122         if (block.jjtGetParent() instanceof ASTIfStatement) {
123             // Else Ifs are their own subnode in AST. So we have to
124             // look a little farther up the tree to find the IF statement
125             Node possibleStatement = statement.getFirstParentOfType(ASTIfStatement.class);
126             while (possibleStatement instanceof ASTIfStatement) {
127                 statement = possibleStatement;
128                 possibleStatement = possibleStatement.getFirstParentOfType(ASTIfStatement.class);
129             }
130         }
131         Map<Node, Integer> thisBranch = blocks.get(statement);
132         if (thisBranch == null) {
133             thisBranch = new HashMap<Node, Integer>();
134             blocks.put(statement, thisBranch);
135         }
136         Integer x = thisBranch.get(block);
137         if (x != null) {
138             thisSize += x;
139         }
140         thisBranch.put(statement, thisSize);
141     }

143     private int processBlocks(Map<Node, Map<Node, Integer>> blocks) {
144         int anticipatedLength = 0;
145         int ifLength = 0;
146         for (Map.Entry<Node, Map<Node, Integer>> entry: blocks.entrySet()) {
147             ifLength = 0;
148             for (Map.Entry<Node, Integer> entry2: entry.getValue().entrySet()) {
149                 Integer value = entry2.getValue();
150                 ifLength = Math.max(ifLength, value.intValue());
151             }
152             anticipatedLength += ifLength;
153         }
154         return anticipatedLength;
155     }

157     private int processAdditive(Node sn) {
158         ASTAdditiveExpression additive = sn.getFirstDescendantOfType(ASTAdditiveExpression.class);
159         if (additive == null) {
160             return 0;
161         }
162         int anticipatedLength = 0;
163         for (int ix = 0; ix < additive.jjtGetNumChildren(); ix++) {
164             Node childNode = additive.jjtGetChild(ix);
165             ASTLiteral literal = childNode.getFirstDescendantOfType(ASTLiteral.class);
166             if (literal != null && literal.getImage() != null) {
167                 anticipatedLength += literal.getImage().length() - 2;
168             }
169         }
170 
171         return anticipatedLength;
172     }

174     private static final boolean isLiteral(String str) {
175         if (str.length() == 0) {
176             return false;
177         }
178         char c = str.charAt(0);
179         return c == '"' || c == '\'';
180     }

182     private int processNode(Node sn) {
183         int anticipatedLength = 0;
184         if ( sn != null ) {
185             ASTPrimaryPrefix xn = sn.getFirstDescendantOfType(ASTPrimaryPrefix.class);
186             if ( xn != null ) {
187                 if (xn.jjtGetNumChildren() != 0 && xn.jjtGetChild(0) instanceof ASTLiteral) {
188                     String str = xn.jjtGetChild(0).getImage();
189                     if (str != null) {
190 	                    if(isLiteral(str)){
191 	                        anticipatedLength += str.length() - 2;
192 	                    } else if(str.startsWith("0x")){
193 	                    anticipatedLength += 1;
194 	                    } else {
195 	                        anticipatedLength += str.length();
196 	                    }
197                     }
198                 }
199             }
200         }
201         return anticipatedLength;
202     }

204     private int getConstructorLength(Node node, int constructorLength) {
205         int iConstructorLength = constructorLength;
206         Node block = node.getFirstParentOfType(ASTBlockStatement.class);
207 
208         if (block == null) {
209             block = node.getFirstParentOfType(ASTFieldDeclaration.class);
210         }
211         if (block == null) {
212             block = node.getFirstParentOfType(ASTFormalParameter.class);
213             if (block != null) {
214                 iConstructorLength = -1;
215             } else {
216             	return DEFAULT_BUFFER_SIZE;
217             }
218         }
219 
220         //if there is any addition/subtraction going on then just use the default.
221         ASTAdditiveExpression exp = block.getFirstDescendantOfType(ASTAdditiveExpression.class);
222         if (exp != null){
223             return DEFAULT_BUFFER_SIZE;
224         }
225         ASTMultiplicativeExpression mult = block.getFirstDescendantOfType(ASTMultiplicativeExpression.class);
226         if (mult != null){
227             return DEFAULT_BUFFER_SIZE;
228         }
229 
230         List<ASTLiteral> literals = block.findDescendantsOfType(ASTLiteral.class);
231         if (literals.isEmpty()) {
232             List<ASTName> name = block.findDescendantsOfType(ASTName.class);
233             if (!name.isEmpty()) {
234                 iConstructorLength = -1;
235             }
236         } else if (literals.size() == 1) {
237             String str = literals.get(0).getImage();
238             if (str == null) {
239                 iConstructorLength = 0;
240             } else if (isLiteral(str)) {
241                 // since it's not taken into account
242                 // anywhere. only count the extra 16
243                 // characters
244                 iConstructorLength = 14 + str.length(); // don't add the constructor's length,
245             } else {
246                 iConstructorLength = Integer.parseInt(str);
247             }
248         } else {
249             iConstructorLength = -1;
250         }
251 
252         if (iConstructorLength == 0) {
253             if (constructorLength == -1) {
254         	iConstructorLength = DEFAULT_BUFFER_SIZE;
255             } else {
256         	iConstructorLength = constructorLength;
257             }
258         }
259 
260         return iConstructorLength;
261     }

264     private int getInitialLength(Node node) {
265 
266     	Node block = node.getFirstParentOfType(ASTBlockStatement.class);
267 
268         if (block == null) {
269             block = node.getFirstParentOfType(ASTFieldDeclaration.class);
270             if (block == null) {
271                 block = node.getFirstParentOfType(ASTFormalParameter.class);
272             }
273         }
274         List<ASTLiteral> literal = block.findDescendantsOfType(ASTLiteral.class);
275         if (literal.size() == 1) {
276             String str = literal.get(0).getImage();
277             if (str != null && isLiteral(str)) {
278                 return str.length() - 2; // take off the quotes
279             }
280         }
281 
282         return 0;
283     }

285     private boolean isAdditive(Node n) {
286         return n.hasDescendantOfType(ASTAdditiveExpression.class);
287     }

289     /**
290      * Locate the block that the given node is in, if any
291      *
292      * @param node
293      *            The node we're looking for a parent of
294      * @return Node - The node that corresponds to any block that may be a
295      *         parent of this object
296      */
297     private Node getFirstParentBlock(Node node) {
298         Node parentNode = node.jjtGetParent();
299 
300         Node lastNode = node;
301         while (parentNode != null && !BLOCK_PARENTS.contains(parentNode.getClass())) {
302             lastNode = parentNode;
303             parentNode = parentNode.jjtGetParent();
304         }
305         if (parentNode instanceof ASTIfStatement) {
306             parentNode = lastNode;
307         } else if (parentNode instanceof ASTSwitchStatement) {
308             parentNode = getSwitchParent(parentNode, lastNode);
309         }
310         return parentNode;
311     }

313     /**
314      * Determine which SwitchLabel we belong to inside a switch
315      *
316      * @param parentNode
317      *            The parent node we're looking at
318      * @param lastNode
319      *            The last node processed
320      * @return The parent node for the switch statement
321      */
322     private static Node getSwitchParent(Node parentNode, Node lastNode) {
323         int allChildren = parentNode.jjtGetNumChildren();
324         ASTSwitchLabel label = null;
325         for (int ix = 0; ix < allChildren; ix++) {
326             Node n = parentNode.jjtGetChild(ix);
327             if (n instanceof ASTSwitchLabel) {
328                 label = (ASTSwitchLabel) n;
329             } else if (n.equals(lastNode)) {
330                 parentNode = label;
331                 break;
332             }
333         }
334         return parentNode;
335     }
