142 	public Object visit(ASTCompilationUnit node, Object data) {
143 		String className = null;
144 		try {
145 			importedOnDemand = new ArrayList<String>();
146 			importedClasses = new HashMap<String, String>();
147 			className = getClassName(node);
148 			if (className != null) {
149 				populateClassName(node, className);
150 			}
151 		} catch (ClassNotFoundException e) {
152 			LOG.log(Level.FINE, "Could not find class " + className + ", due to: " + e.getClass().getName() + ": " + e.getMessage());
153 		} catch (NoClassDefFoundError e) {
154 			LOG.log(Level.WARNING, "Could not find class " + className + ", due to: " + e.getClass().getName() + ": " + e.getMessage());
155 		} catch (ClassFormatError e) {
156 			LOG.log(Level.WARNING, "Could not find class " + className + ", due to: " + e.getClass().getName() + ": " + e.getMessage());
157 		} finally {
158 			populateImports(node);
159 		}
160 		return super.visit(node, data);
161 	}

214 	@Override
215 	public Object visit(ASTName node, Object data) {
216 		/*
217 		 * Only doing this for nodes where getNameDeclaration is null this means
218 		 * it's not a named node, i.e. Static reference or Annotation Doing this
219 		 * for memory - TODO: Investigate if there is a valid memory concern or
220 		 * not
221 		 */
222 		if (node.getNameDeclaration() == null) {
223 			// Skip these scenarios as there is no type to populate in these cases:
224 			// 1) Parent is a PackageDeclaration, which is not a type
225 			// 2) Parent is a ImportDeclaration, this is handled elsewhere.
226 			if (!(node.jjtGetParent() instanceof ASTPackageDeclaration || node.jjtGetParent() instanceof ASTImportDeclaration)) {
227 				String name = node.getImage();
228 				if (name.indexOf('.') != -1) {
229 					name = name.substring(0, name.indexOf('.'));
230 				}
231 				populateType(node, name);
232 			}
233 		} else {
234 			// Carry over the type from the declaration
235 			if (node.getNameDeclaration().getNode() instanceof TypeNode) {
236 				node.setType(((TypeNode)node.getNameDeclaration().getNode()).getType());
237 			}
238 		}
239 		return super.visit(node, data);
240 	}

467 	@Override
468 	public Object visit(ASTLiteral node, Object data) {
469 		super.visit(node, data);
470 		if (node.jjtGetNumChildren() != 0) {
471 			rollupTypeUnary(node);
472 		} else {
473 			if (node.isIntLiteral()) {
474 				String image = node.getImage();
475 				if (image.endsWith("l") || image.endsWith("L")) {
476 					populateType(node, "long");
477 				} else {
478 					try {
479 						Integer.decode(image);
480 						populateType(node, "int");
481 					} catch (NumberFormatException e) {
482 						// Bad literal, 'long' requires use of 'l' or 'L' suffix.
483 					}
484 				}
485 			} else if (node.isFloatLiteral()) {
486 				String image = node.getImage();
487 				if (image.endsWith("f") || image.endsWith("F")) {
488 					populateType(node, "float");
489 				} else if (image.endsWith("d") || image.endsWith("D")) {
490 					populateType(node, "double");
491 				} else {
492 					try {
493 						Double.parseDouble(image);
494 						populateType(node, "double");
495 					} catch (NumberFormatException e) {
496 						// Bad literal, 'float' requires use of 'f' or 'F' suffix.
497 					}
498 				}
499 			} else if (node.isCharLiteral()) {
500 				populateType(node, "char");
501 			} else if (node.isStringLiteral()) {
502 				populateType(node, "java.lang.String");
503 			} else {
504 				throw new IllegalStateException("PMD error, unknown literal type!");
505 			}
506 		}
507 		return data;
508 	}

547 	// Roll up the type based on type of the first child node.
548 	private void rollupTypeUnary(TypeNode typeNode) {
549 		if (typeNode instanceof Node) {
550 			Node node = (Node)typeNode;
551 			if (node.jjtGetNumChildren() >= 1) {
552 				Node child = node.jjtGetChild(0);
553 				if (child instanceof TypeNode) {
554 					typeNode.setType(((TypeNode)child).getType());
555 				}
556 			}
557 		}
558 	}

560 	// Roll up the type based on type of the first child node using Unary Numeric Promotion per JLS 5.6.1
561 	private void rollupTypeUnaryNumericPromotion(TypeNode typeNode) {
562 		if (typeNode instanceof Node) {
563 			Node node = (Node)typeNode;
564 			if (node.jjtGetNumChildren() >= 1) {
565 				Node child = node.jjtGetChild(0);
566 				if (child instanceof TypeNode) {
567 					Class<?> type = ((TypeNode)child).getType();
568 					if (type != null) {
569 						if ("byte".equals(type.getName()) || "short".equals(type.getName())
570 								|| "char".equals(type.getName())) {
571 							populateType(typeNode, "int");
572 						} else {
573 							typeNode.setType(((TypeNode)child).getType());
574 						}
575 					}
576 				}
577 			}
578 		}
579 	}

581 	// Roll up the type based on type of the first and second child nodes using Binary Numeric Promotion per JLS 5.6.2
582 	private void rollupTypeBinaryNumericPromotion(TypeNode typeNode) {
583 		if (typeNode instanceof Node) {
584 			Node node = (Node)typeNode;
585 			if (node.jjtGetNumChildren() >= 2) {
586 				Node child1 = node.jjtGetChild(0);
587 				Node child2 = node.jjtGetChild(1);
588 				if (child1 instanceof TypeNode && child2 instanceof TypeNode) {
589 					Class<?> type1 = ((TypeNode)child1).getType();
590 					Class<?> type2 = ((TypeNode)child2).getType();
591 					if (type1 != null && type2 != null) {
592 						// Yeah, String is not numeric, but easiest place to handle it, only affects ASTAdditiveExpression
593 						if ("java.lang.String".equals(type1.getName()) || "java.lang.String".equals(type2.getName())) {
594 							populateType(typeNode, "java.lang.String");
595 						} else if ("boolean".equals(type1.getName()) || "boolean".equals(type2.getName())) {
596 							populateType(typeNode, "boolean");
597 						} else if ("double".equals(type1.getName()) || "double".equals(type2.getName())) {
598 							populateType(typeNode, "double");
599 						} else if ("float".equals(type1.getName()) || "float".equals(type2.getName())) {
600 							populateType(typeNode, "float");
601 						} else if ("long".equals(type1.getName()) || "long".equals(type2.getName())) {
602 							populateType(typeNode, "long");
603 						} else {
604 							populateType(typeNode, "int");
605 						}
606 					} else if (type1 != null || type2 != null) {
607 						// If one side is known to be a String, then the result is a String
608 						// Yeah, String is not numeric, but easiest place to handle it, only affects ASTAdditiveExpression
609 						if (type1 != null && "java.lang.String".equals(type1.getName())
610 								|| type2 != null && "java.lang.String".equals(type2.getName())) {
611 							populateType(typeNode, "java.lang.String");
612 						}
613 					}
614 				}
615 			}
616 		}

620 
621 		String qualifiedName = className;
622 		Class<?> myType = PRIMITIVE_TYPES.get(className);
623 		if (myType == null && importedClasses != null) {
624 			if (importedClasses.containsKey(className)) {
625 				qualifiedName = importedClasses.get(className);
626 			} else if (importedClasses.containsValue(className)) {
627 				qualifiedName = className;
628 			}
629 			if (qualifiedName != null) {
630 				try {
631 					/*
632 					 * TODO - the map right now contains just class names. if we
633 					 * use a map of classname/class then we don't have to hit
634 					 * the class loader for every type - much faster
635 					 */
636 					myType = pmdClassLoader.loadClass(qualifiedName);
637 				} catch (ClassNotFoundException e) {
638 					myType = processOnDemand(qualifiedName);
639 				} catch (NoClassDefFoundError e) {
640 					myType = processOnDemand(qualifiedName);
641 				} catch (ClassFormatError e) {
642 					myType = processOnDemand(qualifiedName);
643 				}
644 			}
645 		}
646 		if (myType != null) {
647 			node.setType(myType);
648 		}

674 		ASTClassOrInterfaceDeclaration classDecl = node.getFirstDescendantOfType(ASTClassOrInterfaceDeclaration.class);
675 		if (classDecl == null) {
