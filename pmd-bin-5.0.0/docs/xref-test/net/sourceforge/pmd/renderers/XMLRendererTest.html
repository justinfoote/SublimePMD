111         RuleContext ctx = new RuleContext();
112         ctx.setReport(report);
113         ctx.setSourceCodeFilename("bar");
114         RuleSet rules = new RuleSet();
115         rules.addRule(rule);
116         p.getSourceCodeProcessor().processSourceCode(new StringReader(TEST2), new RuleSets(rules), ctx);
117         Element root = parseRootElement(report);
