28      /**
29       * Run the PMD command line tool, i.e. call PMD.main().
30       * 
31       * @param commandLine
32       */
33      private void runPmd(String commandLine) {
34      	String[] args;
35      	args = commandLine.split("\\s");
36  
37      	File f = null;
38      	try {
39      		f = File.createTempFile("pmd", ".txt");
40      		int n = args.length;
41      		String[] a = new String[n + 2];
42      		System.arraycopy(args, 0, a, 0, n);
43      		a[n] = "-reportfile";
44      		a[n + 1] = f.getAbsolutePath();
45      		args = a;
46  
47      		PMD.run(args);
48  
49      		// FIXME: check that output doesn't have parsing errors
50      	} catch (IOException ioe) {
51      		fail("Problem creating temporary file: " + ioe.getLocalizedMessage());
52      	} finally {
53      		if (f != null) f.delete();
54      	}
55      }

62      /**
63       * Run PMD using the command lines found in PMD_CONFIG_FILE.
64       */
65      @Test
66      public void testResourceFileCommands() {
67  
68  	InputStream is = getClass().getResourceAsStream(PMD_CONFIG_FILE);
69  
70  	if (is != null) {
71  	    try {
72  		BufferedReader r = new BufferedReader(new InputStreamReader(is));
73  		String l;
74  		while ((l = r.readLine()) != null) {
75  		    l = l.trim();
76  		    if (l.length() == 0 || l.charAt(0) == '#') {
77  			continue;
78  		    }
79  
80  		    runPmd(l);
81  		}
82  		r.close();
83  	    } catch (IOException ioe) {
84  		fail("Problem reading config file: " + ioe.getLocalizedMessage());
85  	    }
86  	} else {
87  	    fail("Missing config file: " + PMD_CONFIG_FILE);
88  	}
89      }
